<script>
// QUESTA È LA TUA "FORMULA EXCEL" TRADOTTA
function eseguiCalcolo() {
    // 1. Prendo i valori inseriti dall'utente
    const quantita = document.getElementById('input_quantita').value;
    const prezzoUnitario = document.getElementById('scelta_prezzo').value;

    // 2. Eseguo l'operazione matematica (es. Moltiplicazione)
    let risultato = quantita * prezzoUnitario;

    // 3. Mostro il risultato nella pagina (formattato bene)
    document.getElementById('output_totale').innerText = risultato.toLocaleString('it-IT');
}
